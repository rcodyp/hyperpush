    **src/routes/+page.server.js** — Homepage: fetch recent packages:
    ```javascript
    export async function load({ fetch }) {
      try {
        const res = await fetch('https://api.packages.meshlang.dev/api/v1/packages');
        if (!res.ok) return { packages: [], error: 'Registry unavailable' };
        const packages = await res.json();
        return { packages };
      } catch {
        return { packages: [], error: 'Failed to fetch packages' };
      }
    }
    ```

    **src/routes/search/+page.server.js** — Search via ?q= query param:
    ```javascript
    export async function load({ fetch, url }) {
      const q = url.searchParams.get('q') || '';
      if (!q.trim()) return { packages: [], query: q };
      try {
        const res = await fetch(`https://api.packages.meshlang.dev/api/v1/packages?q=${encodeURIComponent(q)}`);
        if (!res.ok) return { packages: [], query: q, error: 'Registry unavailable' };
        const packages = await res.json();
        return { packages, query: q };
      } catch {
        return { packages: [], query: q, error: 'Search failed' };
      }
    }
    ```

        const res = await fetch(`https://api.packages.meshlang.dev/api/v1/packages/${params.name}`);
