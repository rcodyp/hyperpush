Cancel handle contract (LOCKED DECISION from CONTEXT.md):
- Http.stream / Http.stream_bytes return a cancel handle (u64 wrapping Arc<AtomicBool>).
- Http.cancel(handle) sets the AtomicBool to true WITHOUT dropping the Arc
  (the stream OS thread still holds a clone — using Arc::from_raw then Arc::into_raw to peek without drop).
- The stream thread polls the AtomicBool each loop iteration and breaks if set.
```rust
// Return cancel handle from mesh_http_stream (returned as i64 / Int in Mesh):
let cancel = Arc::new(AtomicBool::new(false));
let cancel_for_thread = cancel.clone();
// Box the Arc so it has a stable address:
let cancel_handle: u64 = Box::into_raw(Box::new(cancel)) as u64;
// Thread uses cancel_for_thread; caller gets cancel_handle.

// mesh_http_cancel implementation — peek without dropping the outer Box:
pub extern "C" fn mesh_http_cancel(cancel_handle: u64) {
    unsafe {
        // Reconstruct reference (NOT Box::from_raw — that would drop):
        let arc = &*(cancel_handle as *const Arc<AtomicBool>);
        arc.store(true, Ordering::SeqCst);
        // The Box<Arc<AtomicBool>> at cancel_handle address is NOT dropped here.
        // It will be dropped when the caller calls a future Http.cancel_free(handle),
        // or on process exit. For now, callers are expected to stop calling cancel
        // after cancellation (handle is single-use for cancellation purpose).
    }
}
```

**Function 1: mesh_http_stream(req_handle: u64, callback_fn: *mut u8, callback_env: *mut u8) -> i64**

CRITICAL: Returns the cancel handle as i64 (NOT Result<HttpResponse, String>). The cancel handle is a `Box<Arc<AtomicBool>>` cast to u64/i64. On network error before streaming starts, return 0 (null handle) — the caller can check the handle is non-zero to detect early failure.

Per CONTEXT.md locked decision: "Http.stream/Http.stream_bytes returns a cancel handle; caller can call Http.cancel(handle)".

```rust
use std::io::Read;
use std::sync::Arc;
use std::sync::atomic::{AtomicBool, Ordering};

#[no_mangle]
pub extern "C" fn mesh_http_stream(
    req_handle: u64,
    callback_fn: *mut u8,
    callback_env: *mut u8,
) -> i64 {
    unsafe {
        let req_data = &*(req_handle as *const MeshRequestData);

        let agent = build_one_shot_agent(req_data.timeout_ms);
        let request = build_request(&agent, req_data);
        match request.http_status_as_error(false).call() {
            Err(_) => {
                // Network error before stream starts — return 0 (null cancel handle)
                0
            }
            Ok(response) => {
                // Create cancel AtomicBool — both stream thread and caller get a clone
                let cancel = Arc::new(AtomicBool::new(false));
                let cancel_for_thread = cancel.clone();
                // Box the Arc for stable address; caller gets this handle
                let cancel_handle = Box::into_raw(Box::new(cancel)) as u64;

                let mut body = response.into_body();
                let fn_ptr = callback_fn;
                let env_ptr = callback_env;

                std::thread::spawn(move || {
                    let mut reader = body.as_reader();
                    let mut buf = vec![0u8; 8192];
                    loop {
                        if cancel_for_thread.load(Ordering::SeqCst) { break; }
                        match reader.read(&mut buf) {
                            Ok(0) => break,
                            Ok(n) => {
                                let chunk = mesh_string_new(buf.as_ptr(), n as u64);
                                let result = if env_ptr.is_null() {
                                    let f: fn(*mut u8) -> *mut u8 = std::mem::transmute(fn_ptr);
                                    f(chunk as *mut u8)
                                } else {
                                    let f: fn(*mut u8, *mut u8) -> *mut u8 = std::mem::transmute(fn_ptr);
                                    f(env_ptr, chunk as *mut u8)
                                };
                                if is_stop_atom(result) { break; }
                            }
                            Err(_) => break,
                        }
                    }
                    // cancel_for_thread Arc dropped here — cancel_handle Arc still alive
                });

                cancel_handle as i64
            }
        }
    }
}
```

#[no_mangle]
pub extern "C" fn mesh_http_cancel(cancel_handle: u64) {
    unsafe {
        if cancel_handle == 0 { return; }
        // Peek at the Arc without taking ownership — do NOT use Box::from_raw here
        // (that would drop the Arc on function exit, potentially freeing it while
        //  the stream thread still holds its own clone).
        let arc_ref = &*(cancel_handle as *const Arc<AtomicBool>);

    }
}
